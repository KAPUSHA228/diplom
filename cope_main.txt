import streamlit as st
import pandas as pd
import numpy as np
import plotly.express as px
import plotly.graph_objects as go
from sklearn.model_selection import train_test_split, StratifiedKFold
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from xgboost import XGBClassifier
from sklearn.metrics import (
    roc_auc_score,
    f1_score,
    precision_score,
    recall_score,
    roc_curve,
    confusion_matrix,
)
from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
from student_risk_pipeline import (
    load_data,
    add_composite_features,
    correlation_analysis,
    preprocess_data,
    select_features,
    cross_validate_models,
    tune_xgboost,
    cluster_students,
    analyze_cluster_profiles,
    generate_batch_explanations,
)

# Настройка страницы
st.set_page_config(
    page_title="Система мониторинга академических рисков",
    page_icon="📊",
    layout="wide"
)

st.title("📊 Система мониторинга академических рисков студентов")
st.markdown("---")

# Боковая панель с настройками
with st.sidebar:
    st.header("⚙️ Настройки")

    # Загрузка данных
    st.subheader("📁 Загрузка данных")
    data_source = st.radio(
        "Источник данных",
        ["Синтетические данные", "Загрузить CSV файл"]
    )

    uploaded_file = None
    if data_source == "Загрузить CSV файл":
        uploaded_file = st.file_uploader(
            "Выберите CSV файл",
            type=['csv'],
            help="Файл должен содержать колонки с признаками и целевую переменную 'risk_flag'"
        )

    st.markdown("---")

    # Настройки анализа
    st.subheader("🔧 Параметры анализа")
    n_clusters = st.slider("Количество кластеров", 2, 6, 3)
    risk_threshold = st.slider("Порог риска", 0.0, 1.0, 0.5, 0.05)
    use_hp_tuning = st.checkbox("Использовать оптимизацию гиперпараметров", value=False)
    n_iter_tuning = st.slider("Итераций для оптимизации", 10, 50, 20) if use_hp_tuning else 20

    st.markdown("---")

    # Кнопка запуска
    run_analysis = st.button("🚀 Запустить анализ", type="primary", use_container_width=True)

# Основная область
if run_analysis:
    with st.spinner("Загрузка данных..."):
        if data_source == "Синтетические данные":
            df = load_data()
            st.success("✅ Синтетические данные загружены")
        elif uploaded_file is not None:
            df = pd.read_csv(uploaded_file)
            st.success("✅ Данные из файла загружены")
        else:
            st.error("❌ Пожалуйста, загрузите CSV файл")
            st.stop()

    # Показываем базовую информацию о данных
    st.subheader("📋 Информация о данных")
    col1, col2, col3, col4 = st.columns(4)
    with col1:
        st.metric("Всего студентов", len(df))
    with col2:
        if 'risk_flag' in df.columns:
            risk_count = df['risk_flag'].sum()
            st.metric("Студентов в группе риска", risk_count)
    with col3:
        if 'risk_flag' in df.columns:
            risk_pct = (df['risk_flag'].sum() / len(df)) * 100
            st.metric("Процент риска", f"{risk_pct:.1f}%")
    with col4:
        st.metric("Признаков", len(df.columns) - 2)  # исключаем student_id и risk_flag

    # Определяем базовые признаки из ТЗ
    with st.spinner("Определение признаков из ТЗ..."):
        base_features = []

        # Признаки из успеваемости
        for col in ['avg_grade', 'grade_std', 'min_grade', 'max_grade', 'n_courses', 'avg_brs']:
            if col in df.columns:
                base_features.append(col)

        # Признаки из анкетирования
        for col in ['satisfaction_score', 'engagement_score', 'workload_perception']:
            if col in df.columns:
                base_features.append(col)

        # Признаки из психологического тестирования
        for col in ['stress_level', 'motivation_score', 'anxiety_score']:
            if col in df.columns:
                base_features.append(col)

        # Признаки из эссе
        for col in ['n_essays', 'avg_essay_grade']:
            if col in df.columns:
                base_features.append(col)

        # Добавляем композитные признаки
        df = add_composite_features(df)

        # Композитные признаки
        composite_features = ['trend_grades', 'grade_stability', 'cognitive_load',
                             'overall_satisfaction', 'psychological_wellbeing', 'academic_activity']

        # Добавляем композитные признаки к списку
        for col in composite_features:
            if col in df.columns and col not in base_features:
                base_features.append(col)

        st.success(f"✅ Определено {len(base_features)} признаков из ТЗ")

    # Корреляционный анализ
    st.subheader("🔗 Корреляционный анализ")
    if 'risk_flag' in df.columns:
        corr = correlation_analysis(df, base_features, 'risk_flag', output_prefix="corr")

        # Визуализация корреляционной матрицы
        fig_corr = px.imshow(
            corr,
            text_auto=".2f",
            aspect="auto",
            color_continuous_scale="RdBu",
            title="Корреляционная матрица признаков"
        )
        st.plotly_chart(fig_corr, width='stretch')

        st.dataframe(corr, width='stretch')

    # Кластеризация
    st.subheader("🎯 Кластеризация студентов")
    with st.spinner("Выполнение кластеризации..."):
        X_for_clustering = df[base_features].fillna(df[base_features].median(numeric_only=True))
        cluster_labels, kmeans_model, scaler_cluster = cluster_students(X_for_clustering, n_clusters=n_clusters)
        df['cluster'] = cluster_labels

        # Анализ профилей кластеров
        cluster_profiles = analyze_cluster_profiles(df, base_features, cluster_col='cluster')

        col1, col2 = st.columns(2)
        with col1:
            st.write("**Профили кластеров:**")
            st.dataframe(cluster_profiles, width='stretch')

        with col2:
            # Визуализация кластеров
            pca = PCA(n_components=2, random_state=42)
            scaler = StandardScaler()
            X_scaled = scaler.fit_transform(X_for_clustering)
            X_pca = pca.fit_transform(X_scaled)

            fig_clusters = px.scatter(
                x=X_pca[:, 0],
                y=X_pca[:, 1],
                color=cluster_labels,
                title=f"Кластеризация студентов (PCA проекция)",
                labels={'x': f'PC1 ({pca.explained_variance_ratio_[0]:.1%} variance)',
                       'y': f'PC2 ({pca.explained_variance_ratio_[1]:.1%} variance)',
                       'color': 'Кластер'}
            )
            st.plotly_chart(fig_clusters, width='stretch')

    # Предобработка данных
    st.subheader("🔄 Предобработка данных")
    with st.spinner("Обработка данных и применение SMOTE..."):
        if 'risk_flag' in df.columns:
            X_res, y_res = preprocess_data(df, base_features, 'risk_flag')
            X_sel, selected_cols = select_features(X_res, y_res, top_n=7, final_n=5)

            st.success(f"✅ Данные обработаны. Отобрано признаков: {len(selected_cols)}")
            st.write("**Отобранные признаки:**", ", ".join(selected_cols))

    # Кросс-валидация
    st.subheader("📊 Кросс-валидация моделей")
    with st.spinner("Выполнение кросс-валидации..."):
        models_for_cv = {
            'LR': LogisticRegression(max_iter=1000),
            'RF': RandomForestClassifier(random_state=42),
            'XGB': XGBClassifier(eval_metric='logloss', random_state=42),
        }
        cv_results = cross_validate_models(X_sel, y_res, models_for_cv, cv_folds=5, scoring='f1')

        cv_df = pd.DataFrame({
            'Модель': list(cv_results.keys()),
            'F1-score (среднее)': [cv_results[m]['mean'] for m in cv_results.keys()],
            'F1-score (std)': [cv_results[m]['std'] for m in cv_results.keys()]
        })
        st.dataframe(cv_df, width='stretch')

    # Разделение на train/test
    X_train, X_test, y_train, y_test = train_test_split(X_sel, y_res, test_size=0.2, random_state=42)

    # Оптимизация гиперпараметров (опционально)
    if use_hp_tuning:
        st.subheader("⚙️ Оптимизация гиперпараметров XGBoost")
        with st.spinner("Оптимизация гиперпараметров..."):
            xgb_tuned, best_params, best_cv_score = tune_xgboost(X_train, y_train, n_iter=n_iter_tuning, cv_folds=3)
            st.success(f"✅ Оптимизация завершена. Лучший F1-score: {best_cv_score:.4f}")
            st.json(best_params)
            models_for_eval = {
                'LR': LogisticRegression(max_iter=1000),
                'RF': RandomForestClassifier(random_state=42),
                'XGB': xgb_tuned,
            }
    else:
        models_for_eval = {
            'LR': LogisticRegression(max_iter=1000),
            'RF': RandomForestClassifier(random_state=42),
            'XGB': XGBClassifier(eval_metric='logloss', random_state=42),
        }

    # Обучение моделей
    st.subheader("🤖 Обучение и оценка моделей")
    with st.spinner("Обучение моделей..."):
        for name, model in models_for_eval.items():
            model.fit(X_train, y_train)

        # Оценка на тестовой выборке
        results = {}
        best_model = None
        best_f1 = 0
        best_name = None

        for name, model in models_for_eval.items():
            preds = model.predict(X_test)
            proba = model.predict_proba(X_test)[:, 1] if hasattr(model, "predict_proba") else preds
            roc_auc = roc_auc_score(y_test, proba)
            f1 = f1_score(y_test, preds)
            prec = precision_score(y_test, preds)
            rec = recall_score(y_test, preds)
            results[name] = {
                'ROC-AUC': roc_auc,
                'F1': f1,
                'Precision': prec,
                'Recall': rec
            }
            if f1 > best_f1:
                best_f1 = f1
                best_model = model
                best_name = name

        # Таблица результатов
        results_df = pd.DataFrame(results).T
        results_df.columns = ['ROC-AUC', 'F1-Score', 'Precision', 'Recall']
        st.dataframe(results_df.style.highlight_max(axis=0), width='stretch')

        st.success(f"✅ Лучшая модель: **{best_name}** (F1-score: {best_f1:.4f})")

    # ROC-кривые
    st.subheader("📈 ROC-кривые")
    fig_roc = go.Figure()
    fig_roc.add_trace(go.Scatter(x=[0, 1], y=[0, 1], mode='lines', name='Random', line=dict(dash='dash')))

    for name, model in models_for_eval.items():
        if hasattr(model, "predict_proba"):
            proba = model.predict_proba(X_test)[:, 1]
        elif hasattr(model, "decision_function"):
            proba = model.decision_function(X_test)
        else:
            continue
        fpr, tpr, _ = roc_curve(y_test, proba)
        fig_roc.add_trace(go.Scatter(x=fpr, y=tpr, mode='lines', name=name))

    fig_roc.update_layout(
        title="ROC-кривые моделей",
        xaxis_title="False Positive Rate",
        yaxis_title="True Positive Rate"
    )
    st.plotly_chart(fig_roc, width='stretch')

    # Confusion Matrix для лучшей модели
    st.subheader("📊 Confusion Matrix (лучшая модель)")
    preds_best = best_model.predict(X_test)
    cm = confusion_matrix(y_test, preds_best)

    fig_cm = px.imshow(
        cm,
        text_auto=True,
        aspect="auto",
        labels=dict(x="Predicted", y="True", color="Count"),
        title=f"Confusion Matrix: {best_name}"
    )
    st.plotly_chart(fig_cm, width='stretch')

    # Feature Importance для RF/XGB
    if hasattr(best_model, "feature_importances_"):
        st.subheader("⭐ Важность признаков")
        importances = best_model.feature_importances_
        indices = np.argsort(importances)[-10:]

        fig_fi = px.bar(
            x=importances[indices],
            y=[selected_cols[i] for i in indices],
            orientation='h',
            title=f"Важность признаков: {best_name}",
            labels={'x': 'Importance', 'y': 'Признак'}
        )
        st.plotly_chart(fig_fi, width='stretch')

    # Детальные объяснения
    st.subheader("💡 Объяснения предсказаний (SHAP)")
    with st.spinner("Генерация объяснений..."):
        X_test_df = pd.DataFrame(X_test, columns=selected_cols)
        detailed_explanations = generate_batch_explanations(
            best_model, X_test_df, selected_cols, threshold=risk_threshold, top_n=5
        )

        for i, det_exp in enumerate(detailed_explanations, 1):
            with st.expander(f"Студент #{det_exp['student_index']} - Риск: {det_exp['risk_probability']:.1%}"):
                st.markdown(det_exp['explanation'])

    # Экспорт результатов
    st.subheader("💾 Экспорт результатов")
    col1, col2 = st.columns(2)

    with col1:
        # CSV с предсказаниями
        predictions_df = pd.DataFrame(X_test, columns=selected_cols)
        predictions_df['risk_pred'] = best_model.predict(X_test)
        predictions_df['risk_probability'] = best_model.predict_proba(X_test)[:, 1]
        csv = predictions_df.to_csv(index=False).encode('utf-8')
        st.download_button(
            label="📥 Скачать предсказания (CSV)",
            data=csv,
            file_name="predictions.csv",
            mime="text/csv"
        )

    with col2:
        # CSV с объяснениями
        explanations_df = pd.DataFrame(detailed_explanations)
        csv_exp = explanations_df.to_csv(index=False).encode('utf-8')
        st.download_button(
            label="📥 Скачать объяснения (CSV)",
            data=csv_exp,
            file_name="explanations.csv",
            mime="text/csv"
        )

else:
    st.info("👈 Используйте боковую панель для настройки параметров и запуска анализа")
    st.markdown("""
    ### Возможности системы:
    - 📊 **Корреляционный анализ** признаков
    - 🎯 **Кластеризация студентов** на группы
    - 🤖 **Сравнение моделей ML** (LR, RF, XGBoost)
    - ⚙️ **Оптимизация гиперпараметров**
    - 📈 **Визуализация метрик** (ROC, Confusion Matrix)
    - 💡 **Объяснения предсказаний** (SHAP)
    - 💾 **Экспорт результатов**
    """)
